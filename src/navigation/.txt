import React, { useEffect, useRef } from 'react';
import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';
import { getFocusedRouteNameFromRoute } from '@react-navigation/native';
import { Platform, Pressable, View, StyleSheet, Dimensions, Animated, Text } from 'react-native';
import Icon from 'react-native-vector-icons/FontAwesome';
import { useTheme } from '../context/ThemeContext';
import ApplicationsScreen from '../screens/main/applications/ApplicationsScreen';
import HomeStackNavigator from './HomeStackNavigator';
import ProfileStackNavigator from './ProfileStackNavigator';
import SavedStackNavigator from './SavedStackNavigator';
import JobReelsStackNavigator from './JobReelsStackNavigator';
import { typography } from '../theme/typography';
import type { MainTabParamList } from './types';

const Tab = createBottomTabNavigator<MainTabParamList>();
const { width } = Dimensions.get('window');
const hideTabBarOnScreens = ['JobDetail'];



const TAB_BAR_WIDTH = width - 40;
const TAB_WIDTH = TAB_BAR_WIDTH / 5; // 5 active tabs

const SlidingIndicator = ({ activeIndex, colors }: any) => {
  const translateX = useRef(new Animated.Value(0)).current;

  useEffect(() => {
    Animated.spring(translateX, {
      toValue: activeIndex * TAB_WIDTH,
      useNativeDriver: true,
      friction: 8,
      tension: 40,
    }).start();
  }, [activeIndex]);

  if (activeIndex === 2) return null;

  return (
    <Animated.View
      style={[
        styles.indicatorContainer,
        {
          width: TAB_WIDTH,
          transform: [{ translateX }],
        },
      ]}>
      {/* This circle creates the notch/cutout effect by matching the screen background */}
      <View style={[styles.notchMask, { backgroundColor: colors.background }]} />

      {/* Side curves to smooth the transition - Hide at edges to prevent overlapping with bar's rounded corners */}
      {activeIndex > 0 && (
        <View style={[styles.sideCurve, styles.sideCurveLeft, { backgroundColor: colors.surface, left: (TAB_WIDTH / 2) - 32 - 19 }]} />
      )}
      {activeIndex < 4 && (
        <View style={[styles.sideCurve, styles.sideCurveRight, { backgroundColor: colors.surface, right: (TAB_WIDTH / 2) - 32 - 19 }]} />
      )}

      <View style={[styles.activeCircle, { backgroundColor: colors.surface }]} />
    </Animated.View>
  );
};

const CustomTabButton = (props: any) => {
  const { children, onPress } = props;

  return (
    <Pressable
      onPress={onPress}
      style={styles.tabButton}
    >
      {children}
    </Pressable>
  );
};

const TabItem = ({ name, focused, colors, label }: any) => {
  const isReels = label === 'Reels';

  return (
    <View style={styles.itemContainer}>
      <View style={isReels ? [styles.reelsIconContainer, { backgroundColor: focused ? colors.primary : colors.primary + '10' }] : null}>
        <Icon
          name={isReels ? 'play-circle' : name}
          size={isReels ? 28 : (focused ? 20 : 18)}
          color={isReels ? (focused ? '#FFFFFF' : colors.primary) : (focused ? colors.primary : colors.textPlaceholder)}
          style={focused && !isReels ? { top: 3 } : {}}
        />
      </View>
      {!focused && !isReels && (
        <Text style={[styles.inactiveLabel, { color: colors.textPlaceholder }]}>
          {label}
        </Text>
      )}
      {isReels && !focused && (
        <Text style={[styles.inactiveLabel, { color: colors.primary, fontWeight: '700' }]}>
          {label}
        </Text>
      )}
    </View>
  );
};

const MainTabNavigator: React.FC = () => {
  const { colors } = useTheme();
  const [activeIndex, setActiveIndex] = React.useState(0);

  return (
    <View style={{ flex: 1 }}>
      <Tab.Navigator
        screenOptions={({ route }) => {
          const routeName = getFocusedRouteNameFromRoute(route);
          const isHidden = hideTabBarOnScreens.includes(routeName as string);


          return {
            headerShown: false,
            tabBarActiveTintColor: colors.primary,
            tabBarInactiveTintColor: colors.textPlaceholder,
            tabBarShowLabel: false,
            tabBarBackground: () => isHidden ? null : (
              <View style={styles.indicatorLayer}>
                <View style={[styles.tabBarBg, { backgroundColor: colors.surface }]} />
                <SlidingIndicator activeIndex={activeIndex} colors={colors} />
              </View>
            ),
            tabBarStyle: {
              display: isHidden ? 'none' : 'flex',
              backgroundColor: 'transparent',
              position: 'absolute',
              bottom: 50,
              marginHorizontal: 20,
              height: 50,
              borderTopWidth: 0,
              elevation: 0,
              paddingBottom: 0,
            },
            tabBarButton: (props) => <CustomTabButton {...props} />,
          };
        }}>
        <Tab.Screen
          name="Home"
          component={HomeStackNavigator}
          options={{
            tabBarIcon: ({ focused }) => (
              <TabItem name="home" focused={focused} colors={colors} label="Home" />
            ),
          }}
          listeners={{
            tabPress: () => setActiveIndex(0),
          }}
        />

        <Tab.Screen
          name="Applications"
          component={ApplicationsScreen}
          options={{
            tabBarIcon: ({ focused }) => (
              <TabItem name="briefcase" focused={focused} colors={colors} label="Apply" />
            ),
          }}
          listeners={{
            tabPress: () => setActiveIndex(1),
          }}
        />

        <Tab.Screen
          name="JobReels"
          component={JobReelsStackNavigator}
          options={{
            tabBarIcon: ({ focused }) => (
              <TabItem name="video-camera" focused={focused} colors={colors} label="Reels" />
            ),
          }}
          listeners={{
            tabPress: () => setActiveIndex(2),
          }}
        />

        <Tab.Screen
          name="Saved"
          component={SavedStackNavigator}
          options={{
            tabBarIcon: ({ focused }) => (
              <TabItem name="heart" focused={focused} colors={colors} label="Saved" />
            ),
          }}
          listeners={{
            tabPress: () => setActiveIndex(3),
          }}
        />

        <Tab.Screen
          name="Profile"
          component={ProfileStackNavigator}
          options={{
            tabBarIcon: ({ focused }) => (
              <TabItem name="user" focused={focused} colors={colors} label="Profile" />
            ),
          }}
          listeners={{
            tabPress: () => setActiveIndex(4),
          }}
        />
      </Tab.Navigator>
    </View>
  );
};

  },
